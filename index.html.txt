    // small helpers & form handler
    document.getElementById('year').textContent = new Date().getFullYear();

    function scrollToId(id){
      document.getElementById(id).scrollIntoView({behavior:'smooth',block:'start'});
    }

    function floQuick(){
      // quick-call: open mailto or you can integrate Calendly / Google Meet link here
      window.location.href = 'mailto:you@example.com?subject=Quick%20call%20request';
    }

    function handleSubmit(e, bottom){
      e.preventDefault();
      const form = e.target;
      const statusId = bottom ? 'leadStatusBottom' : 'leadStatus';
      const status = document.getElementById(statusId);
      status.textContent = 'Sending...';

      // Simple client-side validation already via required attributes.
      // Replace this behavior with a real API endpoint, Formspree, or Zapier webhook.
      const formData = new FormData(form);
      // Minimal fallback: open user's email client prefilled (mailto). For production: POST to server.
      const to = 'you@example.com';
      const subject = encodeURIComponent('New lead from website: ' + (formData.get('name') || formData.get('bname') || ''));
      const bodyParts = [];
      for (const [k,v] of formData.entries()){
        bodyParts.push(k + ': ' + v);
      }
      const body = encodeURIComponent(bodyParts.join('\\n'));
      // open mailto in new tab/window
      const mailto = `mailto:${to}?subject=${subject}&body=${body}`;
      window.open(mailto,'_blank');

      status.textContent = 'Opened mail client — or replace with your API endpoint for automated leads.';
      setTimeout(()=> status.textContent = '',4000);
      return false;
    }
